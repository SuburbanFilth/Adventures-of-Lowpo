82  component->angularAcc += component->torqueAccumulator * component->invInertiaTensor;
83  // velocity update
84  component->velocity += component->acceleration * dt;
85  component->angularVel += component->angularAcc * dt;

89  component->orientation.y += 0.5f * component->orientation.y * component->angularVel.y * dt;
90  component->orientation.z += 0.5f * component->orientation.z * component->angularVel.z * dt;

111 void PhysicsSystem::Solve(std::vector<std::unique_ptr<Entity>>& entities, std::vector<std::shared_ptr<Collision>>& collisions, std::unordered_map<int, int>& idToIndexMap)
112 {
113  float ELASTICITY = .1f;
114  for (int i = 0; i < collisions.size(); i++)
115  {
116  std::shared_ptr<Collision> collision = collisions[i];
117  int firstEntityIndex = idToIndexMap[collision->first];
118  int secondEntityIndex = idToIndexMap[collision->second];
119 
120  PhysicsComponent* first = entities[firstEntityIndex]->GetComponent<PhysicsComponent>(ComponentType::Physics);
121  PhysicsComponent* second = entities[secondEntityIndex]->GetComponent<PhysicsComponent>(ComponentType::Physics);
122 
123  std::shared_ptr<Collider> firstCollider = collision->firstCollider;
124  std::shared_ptr<Collider> secondCollider = collision->secondCollider;
125 
126  for (int j = 0; j < collision->contacts.size(); j++)
127  {
128  Contact contact = collision->contacts[j];
129  glm::vec3 normal = contact.contactNormal;
130  glm::vec3 rA = contact.contactPoint - firstCollider->center;
131  glm::vec3 rB = contact.contactPoint - secondCollider->center;
132  glm::vec3 vA = first->velocity;
133  glm::vec3 wA = first->angularVel;
134  glm::vec3 vB = second->velocity;
135  glm::vec3 wB = second->angularVel;
136  float invMassA = first->inverseMass;
137  float invMassB = second->inverseMass;
138  glm::mat3 invInertiaTensorA = first->invInertiaTensor;
139  glm::mat3 invInertiaTensorB = second->invInertiaTensor;
140 
141  /*
142  j = nominator / denominator
143  nominator = -(1 + e) * n * ( (vA + wA x rA) - (vB + wB x rB) )
144  denominator = 1/mA + 1/mB + n * ( (invIA (rA x n) x rA) + (invIB (rB x n) x rB) )
145  ----------- --------------------- ---------------------
146  / / /
147  / / /
148  invMassSum angularPart1 angularPart2
149  */
150  glm::vec3 preRelVelocity = vA + glm::cross(wA, rA) - vB - glm::cross(wB, rB);
151  float nominator = glm::dot((-1.0f - ELASTICITY) * preRelVelocity, normal);
152  float invMassSum = invMassA + invMassB;
153  glm::vec3 angularPart1 = glm::cross(invInertiaTensorA * glm::cross(rA, normal), rA);
154  glm::vec3 angularPart2 = glm::cross(invInertiaTensorB * glm::cross(rB, normal), rB);
155  float angularFinal = glm::dot(angularPart1 + angularPart2, normal);
156  float impulse = nominator / (invMassSum + angularFinal);
157 
158  if (firstCollider->dynamicType != DynamicType::Static)
159  {
160  first->velocity = vA + impulse * normal * invMassA;
161  first->angularVel = wA + invInertiaTensorA * glm::cross(rA, impulse * normal);
162  }
163  if (secondCollider->dynamicType != DynamicType::Static)
164  {
165  second->velocity = vB - impulse * normal * invMassB;
166  second->angularVel = wB - invInertiaTensorB * glm::cross(rB, impulse * normal);
167  }
168  }
169  }
170 }
